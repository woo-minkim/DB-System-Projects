CREATE TABLE Agent (agent_id INT PRIMARY KEY, name VARCHAR(100), phone VARCHAR(20));
CREATE TABLE Buyer (buyer_id INT PRIMARY KEY, name VARCHAR(100), phone VARCHAR(20));
CREATE TABLE Seller (seller_id INT PRIMARY KEY, name VARCHAR(100), phone VARCHAR(20));
CREATE TABLE Property (property_id INT PRIMARY KEY, address VARCHAR(255), status VARCHAR(50), listing_price DECIMAL(15, 2), description TEXT, bedroom_count INT, bathroom_count INT, agent_id INT, seller_id INT, school_district VARCHAR(100), listed_date DATE, property_type VARCHAR(50), Image_url VARCHAR(255), FOREIGN KEY (agent_id) REFERENCES Agent(agent_id), FOREIGN KEY (seller_id) REFERENCES Seller(seller_id));
CREATE TABLE Transaction (transaction_id INT PRIMARY KEY, PropertyID INT, buyer_id INT, Selling_agent_id INT, buying_agent_id INT, sale_price DECIMAL(15, 2), sale_date DATE, FOREIGN KEY (PropertyID) REFERENCES Property(property_id), FOREIGN KEY (buyer_id) REFERENCES Buyer(buyer_id), FOREIGN KEY (Selling_agent_id) REFERENCES Agent(agent_id), FOREIGN KEY (buying_agent_id) REFERENCES Agent(agent_id));
CREATE TABLE Represent (agent_id INT, property_id INT, PRIMARY KEY (agent_id, property_id), FOREIGN KEY (agent_id) REFERENCES Agent(agent_id), FOREIGN KEY (property_id) REFERENCES Property(property_id));
CREATE TABLE Sell (seller_id INT, property_id INT, PRIMARY KEY (seller_id, property_id), FOREIGN KEY (seller_id) REFERENCES Seller(seller_id), FOREIGN KEY (property_id) REFERENCES Property(property_id));
CREATE TABLE Buy (buyer_id INT, transaction_id INT, PRIMARY KEY (buyer_id, transaction_id), FOREIGN KEY (buyer_id) REFERENCES Buyer(buyer_id), FOREIGN KEY (transaction_id) REFERENCES Transaction(transaction_id));
